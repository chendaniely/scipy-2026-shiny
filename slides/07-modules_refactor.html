Refactor Your App Into a Module

Where we’re starting

The tips dashboard’s sidebar today: one slider (total_bill), one checkbox group (time), one @reactive.calc filtered_data(). All hand-written, all tied to the tips dataframe specifically.

Step 1 — identify the reusable part

  • The sidebar UI (slider + checkbox group)
  • The filtering logic (@reactive.calc)
  • Everything else (value boxes, plots, tables) stays in the app — it’s specific to how this app wants to display the filtered data

Step 2 — write the module

We already built this in the last section: filter_panel(input, output, session, data, columns) — pass in any dataframe and any list of columns, get back {"mask": <reactive.calc>}.

Step 3 — swap the sidebar for the module

Before:

with ui.sidebar(open="desktop"):
    ui.input_slider(id="slider", label="Bill amount", ...)
    ui.input_checkbox_group(id="checkbox_group", label="Food service", ...)

@reactive.calc
def filtered_data():
    idx1 = tips.total_bill.between(...)
    idx2 = tips.time.isin(...)
    return tips[idx1 & idx2]

After:

with ui.sidebar():
    tips_filter = filter_panel(
        "tips_filter", data=tips, columns=["total_bill", "time"]
    )

Everywhere the app used filtered_data(), it now uses tips.loc[tips_filter["mask"]()].

## file: filter_module_express.py
"""Reusable Shiny Express module: same behavior as filter_module_core.py,
using the Express `@module` decorator (one function is both UI and server)."""

import pandas as pd
from shiny import reactive
from shiny.express import module, render, ui


def _make_filters(data, columns):
    filters = {}
    for col in columns:
        if pd.api.types.is_numeric_dtype(data[col]):
            lo, hi = float(data[col].min()), float(data[col].max())
            filters[col] = {
                "kind": "range",
                "component": ui.input_slider(
                    f"filter_{col}",
                    f"Range for {col}",
                    min=lo,
                    max=hi,
                    value=[lo, hi],
                ),
            }
        else:
            choices = sorted(data[col].unique())
            filters[col] = {
                "kind": "isin",
                "component": ui.input_checkbox_group(
                    f"filter_{col}",
                    f"Select {col}",
                    choices=choices,
                    selected=choices,
                ),
            }
    return filters


@module
def filter_panel(input, output, session, data, columns):
    ui_filters = _make_filters(data, columns)

    @render.ui
    def filters():
        return [ui_filters[col]["component"] for col in columns]

    @reactive.calc
    def mask():
        m = pd.Series(True, index=data.index)
        for col in columns:
            if ui_filters[col]["kind"] == "range":
                lo, hi = input[f"filter_{col}"]()
                m = m & data[col].between(lo, hi)
            else:
                m = m & data[col].isin(input[f"filter_{col}"]())
        return m

    return {"mask": mask}
